\begin{problem}[points = 17]

(1) 求 $p_3, p_4$ (用 $p$ 表示);

(2) 若 $\dfrac{p_4-p_3}{q_4-q_3}=4$，求 $p$;

(3) 证明: 对于任意正整数 $m$，$p_{2m+1}-q_{2m+1} < p_{2m}-q_{2m} < p_{2m+2}-q_{2m+2}$.
\end{problem}
