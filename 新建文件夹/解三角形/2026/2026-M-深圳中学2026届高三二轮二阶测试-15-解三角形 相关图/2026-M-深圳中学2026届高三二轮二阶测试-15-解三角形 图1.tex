\begin{tikzpicture}[scale=1.2]
    % 1. 定义坐标
    \coordinate (B) at (0,0);
    \coordinate (D) at (3,0);
    \coordinate (C) at (6,0);
    % 使 AB = AD，利用对称性 A 在 B 和 D 的中垂线上
    \coordinate (A) at (1.5, 2.5);

    % 2. 绘制三角形的边
    \draw[thick] (A) -- (B) -- (C) -- cycle;
    \draw[thick] (A) -- (D);

    % 3. 标注顶点
    \node[above] at (A) {A};
    \node[below left] at (B) {B};
    \node[below right] at (C) {C};
    \node[below] at (D) {D};

    % 4. 绘制角度与标注
    % 绘制角 \beta
    \pic [draw, angle radius=0.5cm, "$\beta$", angle eccentricity=1.4] {angle = C--B--A};
    % 绘制角 \alpha
    \pic [draw, angle radius=0.6cm, "$\alpha$", angle eccentricity=1.3] {angle = D--A--C};
\end{tikzpicture}
