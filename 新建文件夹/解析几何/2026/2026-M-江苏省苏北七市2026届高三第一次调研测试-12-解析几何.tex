\begin{problem}{2023}{G}{新课标I}{12}{解析几何}
已知直线$l_1:x-2y+1=0$与直线$l_2:ax+y-1=0$平行，则实数$a$的值为\underline{\hspace{4em}}。
\end{problem}

\begin{answer}
-\dfrac{1}{2}
\end{answer}

\begin{solutions}
两直线平行$\Leftrightarrow$斜率相等。$l_1$斜率$k_1=\dfrac{1}{2}$，$l_2$斜率$k_2=-a$，故$-a=\dfrac{1}{2}\Rightarrow a=-\dfrac{1}{2}$。
\end{solutions}
