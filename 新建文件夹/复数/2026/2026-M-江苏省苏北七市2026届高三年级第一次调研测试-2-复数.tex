\begin{problem}{2026}{M}{高三第一次调研测试}{2}{复数}
已知复数 $z=\dfrac{2}{1+i}$，则 $\overline{z}=$
\begin{choices}
\choice{{$1-i$}}
\choice{{$1+i$}}
\choice{{$-1-i$}}
\choice{{$-1+i$}}
\end{choices}
\end{problem}

\begin{answer}
B
\end{answer}

\begin{solutions}
$z=\dfrac{2}{1+i}=\dfrac{2(1-i)}{(1+i)(1-i)}=\dfrac{2(1-i)}{2}=1-i$，  
故 $\overline{z}=1+i$。
\end{solutions}
